  course_id text NOT NULL,
  bidding_window text NOT NULL,

  course_code text,

  section text,

CREATE TABLE public.profiles (
  id uuid NOT NULL,
  display_name text NOT NULL,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  is_admin boolean NOT NULL DEFAULT false,
  CONSTRAINT profiles_pkey PRIMARY KEY (id),
  CONSTRAINT profiles_id_fkey FOREIGN KEY (id) REFERENCES auth.users(id)
);
CREATE TABLE public.predictions (
  id bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  user_id uuid NOT NULL,
  course_id text NOT NULL,
  bidding_window text NOT NULL,
  target text NOT NULL CHECK (target = ANY (ARRAY['median'::text, 'min'::text])),
  mode text NOT NULL CHECK (mode = ANY (ARRAY['training'::text, 'live'::text])),
  predicted_value numeric NOT NULL CHECK (predicted_value >= 0::numeric),
  actual_value numeric,
  error_pct numeric,
  score numeric,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  course_code text,
  counted boolean NOT NULL DEFAULT false,
  school_department text,
  CONSTRAINT predictions_pkey PRIMARY KEY (id),
  CONSTRAINT predictions_user_id_fkey FOREIGN KEY (user_id) REFERENCES public.profiles(id),
  CONSTRAINT fk_predictions_section FOREIGN KEY (course_id) REFERENCES public.bidding_table_info(course_id),
  CONSTRAINT fk_predictions_section FOREIGN KEY (bidding_window) REFERENCES public.bidding_table_info(bidding_window)
);
